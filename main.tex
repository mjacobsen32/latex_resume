% Custom commands
\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-2pt}\nolinebreak}
  }
}

% Reference entry command
\newcommand{\referenceEntry}[6]{
  \vspace{0.8em}
  \noindent
  \textbf{#1} \\ % Full name
  #2 at #3 \\   % Position at Company
  Phone: #4 \\  % Phone number
  Email: #5\ifthenelse{\equal{#6}{}}{}{ / #6} \\ % Primary email [+ secondary if exists]
}

% Sub Heading with 2 rows of test {company, location} {Dates}
% {position (in bold)} {years} {description} {right side text}
\newcommand{\resumeSubheading}[5]{
  \item
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}
      #1 & {\color{dark-color}\scriptsize #2}\vspace{1pt}\\ % top row of resume entry
      {\textbf{#3} {#4}} & {\color{dark-color} \scriptsize #5}\\ % second row of resume entry
    \end{tabular*}\vspace{-4pt}
}

% SubHeader with 1 row of text for additional positions at the same company
% {position} {years} {extra text}
\newcommand{\resumeSubheadingTwo}[3]{
  \item
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}
      {\textbf{#1} {#2}} & {\color{dark-color} \scriptsize #3}\\
    \end{tabular*}\vspace{-4pt}
}
